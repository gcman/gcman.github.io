def bs(arr, l, r, x):
	while l <= r:
		mid = l + (r - l)//2;
		# Is N in the array?
		if arr[mid] == x:
			# The answer must be less than N
			return arr[mid-1]
		elif arr[mid] < x:
			l = mid + 1
		else:
			r = mid - 1
	return arr[r]
