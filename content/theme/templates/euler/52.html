# Check sorted strings
def is_perm(arr):
	return all([sorted(arr[0]) == sorted(x) for x in arr])

N,K = map(int,input().split())
# We are given N >= 125874

		# Generate multiples
		S = [str(i*j) for j in range(1,K+1)]
		# Check that all multiples are permutations
